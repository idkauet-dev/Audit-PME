    function sendToWebhook(){
    var btn=document.getElementById('btn-send');
    if(!btn) return;
    var blocks=["Stratégie","Pilotage","Processus","RH","Outils","Fournisseurs","Finance","Expérience","Fidélisation","Amélioration"];
    var scores={};
    blocks.forEach(function(b){scores[b]=blockScore(b);});
    var psA=phaseScore('A'),psB=phaseScore('B'),psC=phaseScore('C');
    var globalScore=Math.round((psA+psB+psC)/3);
    var reponses=[];
    questions.forEach(function(q,i){
    var v=ans[i];
    var labelMap={g:'Vert — '+q.g,o:'Orange — '+q.o,r:'Rouge — '+q.r};
    reponses.push({index:i+1,id:q.id,phase:q.phase,bloc:q.block,question:q.text,valeur:v||'',reponse_choisie:v?labelMap[v]:'Sans réponse',commentaire:comments[i]||''});
    });
    var payload={
    meta:{source:'AutoEvaluation - FGBX',date:new Date().toISOString()},
    entreprise:{nom:co.n,secteur:co.sa,taille:co.sz,email:co.e,telephone:co.p},
    scores:{global:globalScore,phase_A:psA,phase_B:psB,phase_C:psC,par_bloc:scores},
    reponses:reponses
    };
    btn.disabled=true;
    btn.innerHTML='<span class="spinner"></span>Envoi en cours...';
    fetch(WEBHOOK_URL,{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify(payload)})
    .then(function(res){
    if(res.ok){btn.innerHTML='✅ Analyse envoyée !';btn.style.background='#2e7d32';showToast('Votre analyse a été envoyée à '+co.e,'ok');}
    else{throw new Error('HTTP '+res.status);}
    })
    .catch(function(err){
    btn.disabled=false;
    btn.innerHTML='📧 Recevoir mon analyse par mail';
    showToast('Erreur d\'envoi — veuillez réessayer ('+err.message+')','err');
    });
    }
